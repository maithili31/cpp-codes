#include <bits/stdc++.h>
using namespace std;

int main () {
    char suite=3;
    switch (suite) {
        case 1:
            printf("diamond\n");
        case 2:
            printf("spade\n");
        default :
            printf("heart\n");
    }
    printf ("i thought one wears a suite\n");
    return 0;
}
